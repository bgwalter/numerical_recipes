    \subsection{Random Number Generator}
    
        \begin{equation}
            x_{j+1} = a x_{j} + c \mod m
            \label{eq:mlcg}
        \end{equation}
        
        \begin{equation}
            \begin{array}{l}
            x_{j+1} = x_{j} \land (x_{j} \ll a) \\
            x_{j+1} = x_{j} \land (x_{j} \gg b) \\
            x_{j+1} = x_{j} \land (x_{j} \ll c)
            \end{array}
            \label{eq:xor_shift}
        \end{equation}
